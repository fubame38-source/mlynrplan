'VIDEO' : hasImage ? 'IMAGE' : 'unknown'; // Link const snapshotUrl = ad.ad_snapshot_url || snap.ad_snapshot_url || `https://www.facebook.com/ads/library/?id=${ad.ad_archive_id || ad.id}`; return { id: ad.ad_archive_id || ad.id || Math.random().toString(36).slice(2), page_name: pageName, start_date: startDate, days_running: daysRunning, power_score: powerScore, platforms: Array.isArray(platforms) ? platforms : [platforms].filter(Boolean), media_type: mediaType, snapshot_url: snapshotUrl, body: body.slice(0, 300), title: title.slice(0, 100), }; }); ads.sort((a, b) => b.power_score - a.power_score); const totalAds = ads.length; const avgDays = totalAds > 0 ?
